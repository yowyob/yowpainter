\section{Conventions générales}

\begin{itemize}[noitemsep]
  \item \textbf{Base URL} : \texttt{http://localhost:8090} (dev) ou URL de production
  \item \textbf{Format} : JSON (\texttt{Content-Type: application/json})
  \item \textbf{Authentification} : \texttt{Authorization: Bearer <JWT>} (sauf routes publiques)
  \item \textbf{Documentation interactive} : \texttt{/swagger-ui.html}
  \item \textbf{Multi-tenant} : header optionnel \texttt{X-Organization-Id: <UUID>}
\end{itemize}

\section{Codes de réponse HTTP}

\begin{table}[h]
  \centering
  \caption{Codes HTTP standards utilisés}
  \label{tab:http-codes}
  \begin{tabular}{@{}ll@{}}
    \toprule
    \textbf{Code} & \textbf{Signification} \\
    \midrule
    200 & Succès \\
    201 & Ressource créée \\
    400 & Requête invalide / validation échouée \\
    401 & Non authentifié / JWT invalide \\
    403 & Accès interdit (rôle insuffisant) \\
    404 & Ressource introuvable \\
    409 & Conflit (ex: e-mail déjà utilisé) \\
    500 & Erreur serveur (Kernel indisponible, etc.) \\
    \bottomrule
  \end{tabular}
\end{table}
