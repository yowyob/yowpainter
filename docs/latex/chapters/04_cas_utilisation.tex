\section{Description détaillée des cas d'utilisation principaux}

\subsection{UC-01 : Inscription Artiste}

\begin{table}[h]
  \centering
  \small
  \begin{tabular}{@{}p{3.5cm}p{10cm}@{}}
    \toprule
    \textbf{Identifiant} & UC-01 \\
    \textbf{Nom} & Inscription Artiste \\
    \textbf{Acteur principal} & Artiste (non authentifié) \\
    \textbf{Préconditions} & Kernel accessible, e-mail valide \\
    \textbf{Postconditions} & Compte créé, statut \texttt{PENDING\_EMAIL} puis \texttt{PENDING\_APPROVAL} \\
    \midrule
    \textbf{Scénario nominal} &
    1. L'artiste envoie \texttt{POST /api/auth/register} avec \texttt{ROLE\_ARTIST}.\\
    & 2. Le backend appelle Kernel sign-up (acteur BUSINESS/ART).\\
    & 3. Création locale \texttt{AppUser} + \texttt{Artist} (slug auto-généré).\\
    & 4. Retour avec message de confirmation e-mail.\\
    & 5. Artiste confirme via \texttt{POST /api/auth/confirm-email}.\\
    & 6. Statut passe à \texttt{PENDING\_APPROVAL}. \\
    \textbf{Extensions} &
    E1 : E-mail déjà utilisé → 409 Conflict.\\
    E2 : Kernel indisponible → 500 avec message explicite. \\
    \bottomrule
  \end{tabular}
\end{table}

    \textbf{Identifiant} & UC-02 \\
    \textbf{Nom} & Validation et provisioning artiste \\
    \textbf{Acteur principal} & Administrateur (\texttt{ROLE\_ADMIN}) \\
    \textbf{Préconditions} & Artiste en \texttt{PENDING\_APPROVAL}, credentials bootstrap valides \\
    \textbf{Postconditions} & Artiste \texttt{ACTIVE}, org Kernel créée, schéma tenant migré \\
    \midrule
    \textbf{Scénario nominal} &
    1. Admin liste \texttt{GET /api/admin/artists/pending}.\\
    & 2. Admin approuve \texttt{POST /api/admin/artists/\{id\}/approve} + MFA.\\
    & 3. \texttt{KernelArtistProvisioningService} : login bootstrap, création org, plan COMMERCE, rôle ORGANIZATION\_ADMIN.\\
    & 4. Migration Liquibase du schéma tenant.\\
    & 5. Statut artiste → \texttt{ACTIVE}. \\
    \bottomrule
  \end{tabular}
\end{table}

\subsection{UC-03 : Achat d'une œuvre (Checkout)}

\begin{table}[h]
  \centering
  \small
  \begin{tabular}{@{}p{3.5cm}p{10cm}@{}}
    \toprule
    \textbf{Identifiant} & UC-03 \\
    \textbf{Nom} & Acheter une œuvre via Mobile Money \\
    \textbf{Acteur principal} & Acheteur (\texttt{ROLE\_BUYER}) \\
    \textbf{Préconditions} & Produit disponible, acheteur authentifié \\
    \textbf{Postconditions} & Commande \texttt{PAID}, wallet artiste crédité (net commission) \\
    \midrule
    \textbf{Scénario nominal} &
    1. Création commande \texttt{POST /api/shop/v1/public/\{slug\}/orders}.\\
    & 2. Checkout \texttt{POST /api/shop/orders/\{id\}/checkout?phoneNumber=...}.\\
    & 3. \texttt{PaymentService} → \kernel \texttt{POST /api/payments/orders} (le kernel pilote le PSP).\\
    & 4. Callback \texttt{POST /api/payment/callback} confirme le paiement.\\
    & 5. Confirmation commande Kernel + crédit wallet + notification. \\
    \bottomrule
  \end{tabular}
\end{table}
